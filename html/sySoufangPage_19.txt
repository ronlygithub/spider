        if (obj.attributes["selected"] == "true") {

            }
        }

    function tijiao(url) {

        if (navigator.userAgent.indexOf("MSIE") > 0) {
            var referLink = document.createElement('a');

        } else {
            window.location.replace(url);
        }
    }
